```typescript
/**
 * Data access abstraction.
 * Currently implemented with JSON files.
 * Designed for drop-in replacement with SQLite/Postgres.
 */
export interface DataLayer {
  // Entries
  listEntries(email: string, category: string): Promise<Entry[]>;
  getEntry(email: string, category: string, id: string): Promise<Entry | null>;
  saveEntry(email: string, category: string, entry: Entry): Promise<Entry>;
  deleteEntry(email: string, category: string, id: string): Promise<boolean>;

  // User Index
  getUserIndex(email: string): Promise<UserIndex | null>;
  saveUserIndex(email: string, index: UserIndex): Promise<void>;

  // Locking
  withLock<T>(key: string, fn: () => Promise<T>): Promise<T>;
}
```
