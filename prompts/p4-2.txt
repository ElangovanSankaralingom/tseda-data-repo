### Step 3: Refactor ShellClient to compose

```typescript
// app/ShellClient.tsx — AFTER refactor
"use client";

import TopNav from "@/components/shell/TopNav";
import Sidebar from "@/components/shell/Sidebar";
import MobileSidebar from "@/components/shell/MobileSidebar";
import NotificationBell from "@/components/shell/NotificationBell";
import UserMenu from "@/components/shell/UserMenu";

export default function ShellClient({ children, session }) {
  const [sidebarOpen, setSidebarOpen] = useState(false);

  return (
    <div className="min-h-screen flex">
      <Sidebar className="hidden lg:block" />
      <MobileSidebar open={sidebarOpen} onClose={() => setSidebarOpen(false)} />
      <div className="flex-1 flex flex-col">
        <TopNav
          onMenuClick={() => setSidebarOpen(true)}
          rightSlot={
            <>
              <NotificationBell />
              <UserMenu session={session} />
            </>
          }
        />
        <main className="flex-1">{children}</main>
      </div>
    </div>
  );
}
```

### State management
If the shell manages global state (notification count, sidebar open, etc.), keep that state in ShellClient and pass down as props. Don't create a context unless absolutely necessary.
