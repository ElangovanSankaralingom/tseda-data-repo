### Step 2: Create shared base component

Create `components/data-entry/adapters/BaseEntryAdapter.tsx`:

This component should:
1. Accept `category` as a prop
2. Look up the schema from the category registry
3. Render the standard editor shell (progress header, status banner, action bar)
4. Render fields by iterating the schema's field definitions
5. Use a `FieldRenderer` component that switches on field type (text, date, select, textarea, file upload, etc.)
6. Handle all shared logic: form state, dirty tracking, save, generate, finalise

```typescript
interface BaseEntryAdapterProps {
  category: CategoryKey;
  entryId?: string;
  isNew?: boolean;
}

export default function BaseEntryAdapter({ category, entryId, isNew }: BaseEntryAdapterProps) {
  const schema = getCategorySchema(category);
  const controller = useCategoryEntryPageController(category, entryId);

  return (
    <CategoryEntryPageShell category={category} controller={controller}>
      <EditorProgressHeader ... />
      <EditorStatusBanner ... />
      
      {schema.fields.map(field => (
        <FieldRenderer
          key={field.name}
          field={field}
          value={controller.formState[field.name]}
          onChange={(val) => controller.setField(field.name, val)}
          disabled={!controller.isEditable}
          error={controller.errors[field.name]}
        />
      ))}

      <EntryActionsBar ... />
    </CategoryEntryPageShell>
  );
}
```

### Step 3: Create FieldRenderer component

Create `components/data-entry/FieldRenderer.tsx`:

```typescript
function FieldRenderer({ field, value, onChange, disabled, error }) {
  switch (field.kind) {
    case 'text': return <TextInput ... />;
    case 'date': return <DateInput ... />;
    case 'select': return <SelectInput options={field.options} ... />;
    case 'textarea': return <TextareaInput ... />;
    case 'number': return <NumberInput ... />;
    case 'file': return <FileUpload ... />;
    default: return <TextInput ... />;
  }
}
```

### Step 4: Simplify each adapter

Each adapter becomes:
```typescript
// components/data-entry/adapters/fdp-attended.tsx
import BaseEntryAdapter from './BaseEntryAdapter';

export default function FdpAttendedAdapter(props) {
  return <BaseEntryAdapter category="fdp-attended" {...props} />;
}
```

If a category has truly unique rendering needs (e.g., a special layout for certain fields), it can override specific fields while still using the base for everything else.
