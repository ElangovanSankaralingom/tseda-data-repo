### 2. Structured Logger with Levels
Enhance `lib/logger.ts`:
```typescript
const LOG_LEVEL = process.env.LOG_LEVEL || (process.env.NODE_ENV === 'production' ? 'info' : 'debug');
const LEVELS = { debug: 0, info: 1, warn: 2, error: 3 };

function shouldLog(level: string): boolean {
  return (LEVELS[level] ?? 0) >= (LEVELS[LOG_LEVEL] ?? 0);
}

export const logger = {
  debug: (data: LogEntry) => shouldLog('debug') && emit('debug', data),
  info: (data: LogEntry) => shouldLog('info') && emit('info', data),
  warn: (data: LogEntry) => shouldLog('warn') && emit('warn', data),
  error: (data: LogEntry) => shouldLog('error') && emit('error', data),
};

function emit(level: string, data: LogEntry) {
  const entry = {
    level,
    ts: new Date().toISOString(),
    requestId: getCurrentRequestId(),
    ...data,
  };
  // In production: structured JSON
  // In dev: human-readable
  if (process.env.NODE_ENV === 'production') {
    console.log(JSON.stringify(entry));
  } else {
    console.log(`[${level.toUpperCase()}] ${data.event || data.message || ''}`);
  }
}
```
