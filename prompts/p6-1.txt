## The Fix

### In the nightly cron handler, add auto-archive logic:

```typescript
async function autoArchiveExpiredEntries() {
  const userDirs = await fs.readdir('.data/users/');
  const categories = ['fdp-attended','fdp-conducted','guest-lectures','case-studies','workshops'];
  let archived = 0;

  for (const userDir of userDirs) {
    for (const cat of categories) {
      const entries = await listEntries(userDir, cat);
      for (const entry of entries) {
        const status = normalizeEntryStatus(entry);
        if (status !== 'GENERATED') continue;
        if (!isEditWindowExpired(entry)) continue;
        // Timer expired — check if valid PDF exists
        if (!entry.pdfGenerated || entry.pdfStale) {
          // No valid PDF → auto-archive
          await archiveEntry(userDir, cat, entry.id, { archiveReason: 'auto_no_pdf' });
          archived++;
        }
        // If valid PDF exists → entry is finalized (read-only), no action needed
      }
    }
  }
  return { archived };
}
```
